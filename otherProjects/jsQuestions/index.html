Write a code that counts the number of Capital B's in a sentence. 
 
              function countBs(bString) {
                var count = 0;
                for (var i = 0; i < bString.length; i++) {
                  if (bString.charAt(i) == 'B') {
                      count = (count + 1);
                  }
                }
                  return count;
              }

Write a range function that takes two arguments, start and end, and returns an array containing all the numbers from start up to (and including) end. 
 
              function range(start, end) {

                var arr = [];
                if (start <= end) { // find out how to do an if NaN
                  for (start; start <= end; start++) {
                    arr.push(start);

                  }
                }
                return arr;
              }

              function sum(array) {
                var sm = 0;
                for (var i = 0; i < array.length; i++) {
                  sm += parseInt(array[i]);
                }
                return sm;

Write a function, reverseArray that takes an array(Apples, Bananas, Oranges) as argument and produces a new array that has the same elements in the inverse order without using .reverse() 
 
              function reverseArray(array){
                var rev = []
                for( var i = array.length-1; i >= 0; i--){

                  rev.push(array[i])
                }
                return rev
              }

Write a function that modifies the array given in order to reverse its elements without creating a new array. 
 
              function arrayReverseInPlace(array) {
                for (var i = 0; i < array.length/2; i++) {
                  var temp = array[i];
                  array[i] = array[array.length - 1 - i];
                  array[array.length-1-i] = temp;

                }
              }
